// function generates password
function GeneratePass(chars,length){
	var res = '';
	var r;
	var i;
	for (i = 1; i <= length; i++) {
		r = Math.floor(Math.random() * chars.length);
		res = res + chars.substring(r,r+1);
	}
	res = res.replace("&","&");
	res = res.replace(">",">");
	res = res.replace("<","<");
	return res;
}
